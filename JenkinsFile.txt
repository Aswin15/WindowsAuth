pipeline {
	agent none
	stages {
		stage('Build') {
			sh 'docker build -f Dockerfile -t ${DOCKER_REGISTRY-}kl-admin:${TAG} --build-arg version=${VERSION}.'	
		}
		stage('Deploy') {
			sh 'docker-compose -f dockercompose.yml --compatibility up -d --no-build --force-recreate --remove-orphans'
		}
	}
}
